            // --- HELPER: NORMALIZE CITY ---
            function normalizeCity(cityName) {
                if (!cityName) return 'Unknown';
                const lower = cityName.toLowerCase().trim();
                if (lower.startsWith('harg') || lower === 'hargeisa' || lower === 'hargaysa' || lower === 'hargiesa' || lower === 'hargeysa') {
                    return 'Hargeisa';
                }
                return cityName.charAt(0).toUpperCase() + cityName.slice(1);
            }

            // --- ALGORITHM: ROUND-ROBIN INTERLEAVER (The Fix) ---
            function interleaveNames(array) {
                if (!array || array.length < 2) return array;

                // 1. Group by First Name (e.g. "Aadan": [...], "Abdi": [...])
                const groups = {};
                array.forEach(item => {
                    const firstName = (item.username || 'Unknown').trim().split(' ')[0].toLowerCase();
                    if (!groups[firstName]) groups[firstName] = [];
                    groups[firstName].push(item);
                });

                // 2. Sort groups by size (Largest groups first)
                // We do this to attack the biggest clusters (like "Aadan") first.
                const sortedKeys = Object.keys(groups).sort((a, b) => groups[b].length - groups[a].length);

                // 3. Round Robin Selection
                // Take one Aadan, then one Abdi, then one Maxamed... repeat until empty.
                const result = [];
                let hasItems = true;

                while (hasItems) {
                    hasItems = false;
                    for (const key of sortedKeys) {
                        if (groups[key].length > 0) {
                            result.push(groups[key].shift()); 
                            hasItems = true;
                        }
                    }
                }

                return result;
            }

                    const sortedCountries = Object.entries(countryCounts).sort((a, b) => b[1] - a[1]).slice(0, 10).reverse();
                    topCountriesReportChart.updateSeries([{ name: 'Voters', data: sortedCountries.map(c => c[1]) }]);
                    topCountriesReportChart.updateOptions({ xaxis: { categories: sortedCountries.map(c => c[0]) } });
                    renderTopCitiesList(true);
                };

                const renderTopCitiesList = (initial = false) => {
                    if (initial) state.topCitiesOffset = 10;
                    const cityCounts = state.filteredVotes.reduce((acc, { city }) => { 
                        const cleanCity = normalizeCity(city);
                        acc[cleanCity] = (acc[cleanCity] || 0) + 1; 
                        return acc; 
                    }, {});
                    const sortedCities = Object.entries(cityCounts).sort((a, b) => b[1] - a[1]);
                    const citiesToShow = sortedCities.slice(0, state.topCitiesOffset);
                    document.getElementById('top-cities-list').innerHTML = citiesToShow.map(([city, count]) => `
                        <li class="flex justify-between items-center py-2 border-b border-gray-100">
                            <span class="font-semibold text-slate-700">${city}</span>
                            <span class="text-slate-500">${count.toLocaleString()} voters</span>
                        </li>`).join('');
                    const loadMoreBtn = document.getElementById('load-more-cities');
                    if (sortedCities.length > state.topCitiesOffset) loadMoreBtn.classList.remove('hidden');
                    else loadMoreBtn.classList.add('hidden');
                };

                // --- ACHIEVEMENTS ---
                const achievementForm = document.getElementById('achievement-upload-form');
                const galleryContainer = document.getElementById('achievements-gallery');

                const renderAchievements = (achievements) => {
                    if(!galleryContainer) return;
                    galleryContainer.innerHTML = achievements.length === 0 ? '<p class="text-slate-500 col-span-full text-center">No images uploaded yet.</p>' : 
                    achievements.map(ach => `
                        <div class="relative group">
                            <img src="${ach.link}" alt="${ach.title}" class="rounded-lg object-cover w-full h-40">
                            <div class="absolute inset-0 bg-black/50 opacity-0 group-hover:opacity-100 transition-opacity flex flex-col justify-end p-2 text-white">
                                <p class="font-bold text-sm truncate">${ach.title}</p>
                                <p class="text-xs">${ach.type}</p>
                            </div>
                            <button data-id="${ach.id}" data-path="${ach.storagePath}" class="delete-achievement-btn absolute top-1 right-1 bg-red-600 text-white p-1 rounded-full opacity-0 group-hover:opacity-100 transition-opacity">
                                <i data-feather="trash-2" class="w-3 h-3"></i>
                            </button>
                        </div>
                    `).join('');
                    feather.replace();
                };

                achievementForm.addEventListener('submit', async (e) => {
                    e.preventDefault();
                    const title = document.getElementById('achievement-title').value;
                    const type = document.getElementById('achievement-type').value;
                    const file = document.getElementById('achievement-file').files[0];
                    const statusEl = document.getElementById('achievement-upload-status');
                    const progressContainer = document.getElementById('upload-progress-container');
                    const progressBar = document.getElementById('upload-progress-bar');
                    const progressText = document.getElementById('upload-progress-text');

                    if (!title || !type || !file) return;

                    const storagePath = `images/${Date.now()}_${file.name}`;
                    const storageRef = storage.ref(storagePath);
                    const uploadTask = storageRef.put(file);
                    
                    progressContainer.classList.remove('hidden');
                    statusEl.textContent = '';
                    
                    uploadTask.on('state_changed', 
                        (snapshot) => {
                            const progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100;
                            progressBar.style.width = progress + '%';
                            progressText.textContent = `Uploading... ${Math.round(progress)}%`;
                        }, 
                        (error) => {
                            statusEl.textContent = "Upload Failed.";
                            statusEl.className = 'text-sm text-center text-red-500 font-bold';
                            progressContainer.classList.add('hidden');
                        }, 
                        async () => {
                            const downloadURL = await uploadTask.snapshot.ref.getDownloadURL();
                            await db.collection('images').add({ title, type, link: downloadURL, storagePath, timestamp: firebase.firestore.FieldValue.serverTimestamp() });
                            statusEl.textContent = "Upload successful!";
                            statusEl.className = 'text-sm text-center text-green-500';
                            achievementForm.reset();
                            setTimeout(() => { statusEl.textContent = ''; progressContainer.classList.add('hidden'); }, 3000);
                        }
                    );
                });

                galleryContainer.addEventListener('click', async (e) => {
                    const button = e.target.closest('.delete-achievement-btn');
                    if (button && confirm('Delete image?')) {
                        try {
                            await db.collection('images').doc(button.dataset.id).delete();
                            await storage.ref(button.dataset.path).delete();
                        } catch (error) { console.error(error); }
                    }
                });

                db.collection('votes').orderBy('timestamp', 'desc').onSnapshot(snap => { state.allVotes = snap.docs.map(d => d.data()); render(); });
                db.collection('pollResponses').orderBy('timestamp', 'desc').onSnapshot(snap => { state.allPollResponses = snap.docs.map(d => d.data()); render(); });
                db.collection('images').orderBy('timestamp', 'desc').onSnapshot(snap => { renderAchievements(snap.docs.map(doc => ({id: doc.id, ...doc.data()}))) });

                // DATE PICKER FIX
                const datePicker = flatpickr("#date-filter", { 
                    mode: "range", 
                    maxDate: null, // Unlimited future/past
                    onChange: (dates) => {
                        state.dateRange = dates.length === 2 ? { start: dates[0], end: new Date(dates[1].setHours(23, 59, 59, 999)) } : { start: null, end: null };
                        state.votersCurrentPage = 1; render();
                    }
                });

                document.querySelectorAll('.date-preset-btn').forEach(btn => btn.addEventListener('click', () => {
                    const end = new Date(), start = new Date();
                    if (btn.dataset.range === 'all') {
                        state.dateRange = { start: null, end: null };
                        datePicker.clear();
                        render();
                        return;
                    }
                    if (btn.dataset.range === 'today') start.setHours(0,0,0,0);
                    else if (btn.dataset.range === 'month') start.setDate(1); // Start of month
                    
                    if (btn.dataset.range !== 'today') start.setHours(0,0,0,0);
                    datePicker.setDate([start, end], true);
                }));

                document.querySelectorAll('.sidebar-link').forEach(link => link.addEventListener('click', (e) => {
                    e.preventDefault();
                    document.querySelectorAll('.sidebar-link').forEach(l => l.classList.remove('active'));
                    link.classList.add('active');
                    const view = link.dataset.view;
                    document.querySelectorAll('.view-content').forEach(v => v.classList.add('hidden'));
                    document.getElementById(`${view}-view`).classList.remove('hidden');
                    let title = view.charAt(0).toUpperCase() + view.slice(1);
                    if (view === "achievements") title = "Achievements Management";
                    document.getElementById('view-title').textContent = title;
                    feather.replace(); 
                }));
                
                document.getElementById('prev-page').addEventListener('click', () => { if (state.votersCurrentPage > 1) { state.votersCurrentPage--; renderVotersTable(); } });
                document.getElementById('next-page').addEventListener('click', () => { if (state.votersCurrentPage * state.votersPerPage < state.filteredVotes.length) { state.votersCurrentPage++; renderVotersTable(); } });
                document.getElementById('search-voters').addEventListener('input', e => { state.searchTerm = e.target.value.toLowerCase(); state.votersCurrentPage = 1; render(); });
                document.getElementById('load-more-cities')?.addEventListener('click', () => { state.topCitiesOffset += 10; renderTopCitiesList(); });
                
                const sidebar = document.querySelector('.sidebar');
                const sidebarToggle = document.getElementById('mobile-sidebar-toggle');
                const sidebarOverlay = document.getElementById('sidebar-overlay');
                sidebarToggle?.addEventListener('click', (e) => { e.stopPropagation(); sidebar.classList.remove('-translate-x-full'); sidebarOverlay.classList.remove('hidden'); });
                sidebarOverlay?.addEventListener('click', () => { sidebar.classList.add('-translate-x-full'); sidebarOverlay.classList.add('hidden'); });
            }
        });
    </script>
</body>
